        var palindrome = function() { 

            var num = palin_input.value; 
        
            var temp = num; 
            var reverse = 0; 
            
            while(num > 0) { 
                digit = num % 10; 
                reverse = reverse * 10 + digit; 
                num = Math.floor(num/10); 
            
            } 
            
            if(temp == reverse) { 
                return "palindrome"; 
            } else { 
                return "not palindrome";
            }
        }
